- 清除缓存 buffer/cache

  - ```shell
    echo "开始清除缓存"
    sync;sync;sync #写入硬盘，防止数据丢失
    sleep 10#延迟10秒
    echo 1 > /proc/sys/vm/drop_caches
    echo 2 > /proc/sys/vm/drop_caches
    echo 3 > /proc/sys/vm/drop_caches
    
    # 创建定时任务
    crontab -e #弹出配置文件
    
    # 添加定时任务执行频率
    #分　 时　 日　 月　 周　 命令
    0 */2 * * * ./cleanCache.sh
    
    # 查看定时任务是否被执行
    cat /var/log/cron | grep cleanCache
    ```
